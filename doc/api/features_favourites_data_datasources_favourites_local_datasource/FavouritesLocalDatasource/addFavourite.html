Future<List<PropertyModel>?> addFavourite(PropertyModel model) async {
  final favs = await listFavourites();
  final index = favs.indexWhere((element) =>
      element.serpapiPropertyDetailsLink == model.serpapiPropertyDetailsLink);
  if (index != -1) return null;

  final newFavs = [...favs, model];
  _sharedPreferencesManager.putString(
      SharedPreferencesManager.favourites, jsonEncode(newFavs));
  return newFavs;
}
