Future<List<PropertyModel>> listFavourites() async {
  final json = _sharedPreferencesManager
      .getString(SharedPreferencesManager.favourites);
  if (json == null) return [];
  return (jsonDecode(json) as List)
      .map((e) => PropertyModel.fromJson(e))
      .toList();
}
